## 1. 시니어 맞춤 도시락

### 1.1 개요
**목적**: 영양 균형 잡힌 시니어 전용 도시락 제공

**핵심 특징**:
- 저염 (나트륨 하루 권장량의 30% 이하)
- 저당 (당뇨 환자 고려)
- 부드러운 식감 (씹기 편하게)
- 영양 균형 (단백질, 식이섬유 강화)
- 적정 칼로리 (400~600kcal/끼)

## 5. 데이터베이스 스키마

```sql
-- 도시락 메뉴
CREATE TABLE meals (
  id UUID PRIMARY KEY,
  name VARCHAR(255),
  category VARCHAR(50), -- 'korean', 'western', 'japanese', 'porridge'
  description TEXT,
  images TEXT[],
  components JSONB, -- [{name: '잡곡밥', amount: '200g', calories: 280}]
  nutrition JSONB, -- {calories, protein, carbs, fat, sodium, sugar, fiber}
  dietary_tags TEXT[], -- ['low_sodium', 'diabetic', 'soft']
  price INTEGER,
  is_available BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 도시락 주문
CREATE TABLE meal_orders (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  meal_id UUID REFERENCES meals(id),
  order_type VARCHAR(20), -- 'one_time', 'subscription'
  delivery_date DATE,
  delivery_time VARCHAR(20), -- 'lunch', 'dinner'
  delivery_address JSONB,
  status VARCHAR(50), -- 'pending', 'confirmed', 'delivered', 'cancelled'
  price INTEGER,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 도시락 구독
CREATE TABLE meal_subscriptions (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  meal_id UUID REFERENCES meals(id),
  frequency VARCHAR(20), -- 'weekdays', 'daily'
  meal_times TEXT[], -- ['lunch', 'dinner']
  start_date DATE,
  end_date DATE,
  status VARCHAR(20), -- 'active', 'paused', 'cancelled'
  created_at TIMESTAMP DEFAULT NOW()
);

-- 식사 기록
CREATE TABLE food_diary (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  meal_time VARCHAR(20), -- 'breakfast', 'lunch', 'dinner', 'snack'
  eaten_at TIMESTAMP,
  foods JSONB, -- [{name: '흰쌀밥', amount: '1공기', calories: 300}]
  photo_url TEXT,
  portion_eaten INTEGER, -- 0-100 (%)
  total_calories INTEGER,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 영양 분석
CREATE TABLE nutrition_analysis (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  date DATE,
  total_calories INTEGER,
  carbs INTEGER,
  protein INTEGER,
  fat INTEGER,
  sodium INTEGER,
  potassium INTEGER,
  fiber INTEGER,
  calcium INTEGER,
  water_ml INTEGER,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 밑반찬
CREATE TABLE banchan_products (
  id UUID PRIMARY KEY,
  name VARCHAR(255),
  category VARCHAR(50), -- 'namul', 'jorim', 'kimchi', 'soup'
  amount VARCHAR(50), -- '300g', '500ml'
  price INTEGER,
  shelf_life_days INTEGER, -- 냉장 보관 일수
  is_available BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 밑반찬 주문
CREATE TABLE banchan_orders (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  items JSONB, -- [{product_id, quantity, price}]
  delivery_date DATE,
  delivery_address JSONB,
  total_price INTEGER,
  status VARCHAR(50),
  created_at TIMESTAMP DEFAULT NOW()
);
```
