const generateVisitSchedule = async (contract: BayadaContract) => {
  const visits = [];
  const startDate = new Date(contract.start_date);
  const endDate = new Date(startDate);
  endDate.setMonth(endDate.getMonth() + contract.contract_duration_months);

  let currentDate = new Date(startDate);

  while (currentDate <= endDate) {
    const dayOfWeek = currentDate.toLocaleDateString('en-US', { weekday: 'short' }).toLowerCase();

    if (contract.visit_days.includes(dayOfWeek)) {
      visits.push({
        contract_id: contract.id,
        professional_id: contract.professional_id,
        visit_date: currentDate.toISOString().split('T')[0],
        scheduled_time: contract.specific_time || '09:00',
        status: 'scheduled'
      });
    }

    currentDate.setDate(currentDate.getDate() + 1);
  }

  await supabase.from('bayada_visits').insert(visits);
};

const BAYADA_API_BASE = process.env.BAYADA_API_URL;
const BAYADA_API_KEY = process.env.BAYADA_API_KEY;

export const matchBayadaProfessional = async (criteria: {
  serviceCategory: string;
  location: string;
  schedule: any;
}) => {
  const response = await fetch(`${BAYADA_API_BASE}/professionals/match`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${BAYADA_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(criteria)
  });

  const data = await response.json();
  return data.professional;
};

export const createBayadaContract = async (contract: any) => {
  const response = await fetch(`${BAYADA_API_BASE}/contracts`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${BAYADA_API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      patient: {
        name: contract.patient_name,
        age: contract.patient_age,
        address: contract.patient_address,
        phone: contract.patient_phone,
        health_conditions: contract.health_conditions
      },
      service: {
        category: contract.service_category_id,
        frequency: contract.frequency_per_week,
        start_date: contract.start_date
      },
      professional_id: contract.professional_id
    })
  });

  const data = await response.json();
  return data;
};
