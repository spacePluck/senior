CREATE OR REPLACE FUNCTION decrement_medication_stock(
  medication_id UUID,
  amount INTEGER DEFAULT 1
)
RETURNS INTEGER AS $$
DECLARE
  new_stock INTEGER;
BEGIN
  UPDATE medications
  SET current_stock = current_stock - amount
  WHERE id = medication_id
  RETURNING current_stock INTO new_stock;

  RETURN new_stock;
END;
