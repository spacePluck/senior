## 1. 가족 대시보드

### 1.1 개요
**목적**: 부모님의 일상과 건강을 한눈에 파악

**접근 방법**:
- 부모님이 가족 초대
- 가족이 앱 다운로드 및 연결
- 역할별 권한 설정 (보기/관리)

## 7. 데이터베이스 스키마

```sql
-- 가족 연결
CREATE TABLE family_connections (
  id UUID PRIMARY KEY,
  senior_id UUID REFERENCES users(id), -- 시니어
  family_id UUID REFERENCES users(id), -- 가족
  relationship VARCHAR(50), -- 'son', 'daughter', 'spouse', 'sibling'
  permission_level VARCHAR(20), -- 'view', 'manage'
  status VARCHAR(20), -- 'active', 'pending', 'inactive'
  invited_at TIMESTAMP,
  connected_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 가족 알림 설정
CREATE TABLE family_notification_settings (
  id UUID PRIMARY KEY,
  family_connection_id UUID REFERENCES family_connections(id),
  medication_alerts BOOLEAN DEFAULT TRUE,
  meal_alerts BOOLEAN DEFAULT TRUE,
  activity_anomaly_alerts BOOLEAN DEFAULT TRUE,
  vital_sign_alerts BOOLEAN DEFAULT TRUE,
  appointment_alerts BOOLEAN DEFAULT TRUE,
  service_alerts BOOLEAN DEFAULT TRUE,
  emergency_alerts BOOLEAN DEFAULT TRUE, -- 항상 true
  quiet_hours_start TIME,
  quiet_hours_end TIME,
  sms_enabled BOOLEAN DEFAULT FALSE,
  email_enabled BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 케어 일지
CREATE TABLE care_logs (
  id UUID PRIMARY KEY,
  senior_id UUID REFERENCES users(id),
  caregiver_id UUID REFERENCES users(id), -- 작성자 (가족)
  log_date DATE,
  log_time TIME,
  activity_type VARCHAR(50), -- 'visit', 'call', 'proxy_task'
  activities TEXT[], -- ['meal', 'cleaning', 'hospital', 'walk']
  senior_condition VARCHAR(20), -- 'good', 'fair', 'poor'
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 대리 작업 기록
CREATE TABLE proxy_actions (
  id UUID PRIMARY KEY,
  senior_id UUID REFERENCES users(id),
  family_id UUID REFERENCES users(id),
  action_type VARCHAR(50), -- 'booking', 'order', 'payment'
  reference_id UUID, -- 예약/주문 ID
  details JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 긴급 알림 이력
CREATE TABLE emergency_alerts (
  id UUID PRIMARY KEY,
  senior_id UUID REFERENCES users(id),
  alert_type VARCHAR(50), -- 'sos', 'fall', 'no_response'
  triggered_at TIMESTAMP,
  resolved_at TIMESTAMP,
  resolution VARCHAR(50), -- 'confirmed_safe', 'false_alarm', '119_called'
  notified_family JSONB, -- [{family_id, notified_at, responded_at}]
  created_at TIMESTAMP DEFAULT NOW()
);

-- 가족 메시지
CREATE TABLE family_messages (
  id UUID PRIMARY KEY,
  senior_id UUID REFERENCES users(id),
  sender_id UUID REFERENCES users(id),
  message_type VARCHAR(20), -- 'text', 'image', 'voice', 'system'
  content TEXT,
  attachments TEXT[],
  sent_at TIMESTAMP DEFAULT NOW(),
  read_by JSONB -- [{user_id, read_at}]
);
```
